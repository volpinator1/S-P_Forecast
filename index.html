def predict(train, test, inputs, model):
    model.fit(train[inputs], train["Target"])
    predictions = model.predict(test[inputs])
    predictions = pd.Series(predictions, index=test.index, name="Predictions")
    combined = pd.concat([test["Target"], predictions], axis=1)
    return combined

#Use 0-10y to predict 11th year, then take 11y of data & predict 12th year,...
def backtest(data, model, inputs, start=2500, step=250): 
    all_predictions = []
    
    for i in range(start, data.shape[0], step):
        train = data.iloc[0:i].copy()
        test = data.iloc[i:(i+step)].copy()
        predictions = predict(train, test, inputs, model)
        all_predictions.append(predictions)
    return pd.concat(all_predictions)
